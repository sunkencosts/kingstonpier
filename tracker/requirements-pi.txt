# Runtime-only dependencies for the tracker on the Raspberry Pi (CPU / ARM64).
#
# Why the +cpu pins + extra index: recent PyTorch aarch64 wheels on PyPI are now
# CUDA-enabled (built for ARM GPU servers like GH200), so a bare `torch>=2.0`
# drags in ~1 GB of nvidia-cudnn/cuda-toolkit the Pi can't use. The `+cpu` local
# version is served only from PyTorch's CPU index below and declares NO nvidia
# deps. These are the same versions dev uses, so the model loads identically.
#
# Install on the Pi with a plain venv (NEVER `uv sync`/pyproject — that's the
# CUDA build for dev):
#   python3 -m venv tracker/.venv
#   tracker/.venv/bin/pip install -r tracker/requirements-pi.txt
#
# Needs 64-bit Raspberry Pi OS (aarch64) with Python 3.10–3.15. `ultralytics` is
# deliberately absent (labelling-only). The trained counter_model.pt is
# git-ignored — copy it to the Pi separately (scp) before starting the worker.

--extra-index-url https://download.pytorch.org/whl/cpu

torch==2.13.0+cpu
torchvision==0.28.0+cpu
numpy>=1.24
pillow>=9.0
requests>=2.28
